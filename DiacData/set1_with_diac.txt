תְּחִיַּת הַלָּשׁוֹן הָעִבְרִית הִיא תַּהֲלִיךְ שֶׁהִתְחוֹלֵל בְּאֵירוֹפָּה וּבְאֶרֶץ יִשְׂרָאֵל בְּסוֹף הַמֵּאָה הַתֵּשַׁע עֶשְׂרֵה וּבִתְחִלַּת הַמֵּאָה הָעֶשְׂרִים, וּבְמַהֲלָכוֹ הָפְכָה הַשָּׂפָה הָעִבְרִית מִלְּשׁוֹן כְּתוּבָה וְלִיטוּרְגִּית, הַמְּשַׁמֶּשֶׁת בְּעִקָּר לִצְרָכִים דָּתִיִּים אוֹ סִפְרוּתִיִּים, לְלָשׁוֹן מְדֻבֶּרֶת, רַב-מַעֲרַכְתִּית וּלְאֻמִּית. תַּהֲלִיךְ זֶה לֹא הָיָה תַּהֲלִיךְ לְשׁוֹנִי גְּרֵידָא, אֶלָּא הוּא הִשְׁתַּבֵּץ בְּמַעֲרֶכֶת רְחָבָה שֶׁל תַּהֲלִיכִים שֶׁחָלוּ בְּעַם יִשְׂרָאֵל בְּאוֹתָהּ תְּקוּפָה, וּבְעִקָּר בַּמַּהְפֵּכָה הַצִּיּוֹנִית וּבַהִתְפַּתְּחוּת הַיִּשּׁוּב הָעִבְרִי הֶחָדָשׁ בְּאֶרֶץ יִשְׂרָאֵל. תַּהֲלִיכִים אֵלֶּה הֵבִיאוּ, בְּסוֹפוֹ שֶׁל דָּבָר, לַהֲקָמַת מְדִינַת יִשְׂרָאֵל כִּמְדִינָה לְאֻמִּית עִבְרִית.

הַשֵּׁם "תְּחִיָּה" אֵינוֹ שֵׁם מְדֻיָּק לְתַהֲלִיךְ זֶה, מִשּׁוּם שֶׁלְּפִי רֹב הַהַגְדָּרוֹת הָעִבְרִית לֹא הָיְתָה לָשׁוֹן מֵתָה לִפְנֵי הַתַּהֲלִיךְ; נַעֲשָׂה בָּהּ שִׁמּוּשׁ נִרְחָב וְנִכְּרוּ בָּהּ תַּהֲלִיכִים שֶׁל הִתְפַּתְּחוּת וְהִשְׁתַּנּוּת גַּם לִפְנֵי תְּחִלַּת הַתַּהֲלִיךְ. עִם זֹאת, הַתַּהֲלִיךְ הוּא יָחִיד בְּמִינוֹ. הַבַּלְשָׁנוּת הַמּוֹדֶרְנִית אֵינָהּ מַכִּירָה מִקְרֶה נוֹסָף שֶׁבּוֹ לָשׁוֹן שֶׁכְּלָל לֹא הָיְתָה מְדֻבֶּרֶת כִּשְׂפַת אֵם הָפְכָה לְלָשׁוֹן לְאֻמִּית הַמְּשַׁמֶּשֶׁת בְּאֹפֶן רַב-מַעֲרַכְתִּי, וְכָל זֹאת בְּתוֹךְ פֶּרֶק זְמַן שֶׁל כַּמָּה עֶשְׂרוֹת שָׁנִים בִּלְבַד (מֻכָּרִים כַּמָּה תַּהֲלִיכִים שֶׁבָּהֶם לָשׁוֹן שֶׁהָיְתָה מְדֻבֶּרֶת בְּפִי מִעוּט הָפְכָה לְלָשׁוֹן כְּלָלִית).

הַתַּהֲלִיךְ הֵבִיא גַּם לְשִׁנּוּיִים לְשׁוֹנִיִּים בַּשָּׂפָה. אַף שֶׁבְּתוֹדַעְתָּם שֶׁל מְחוֹלְלֵי הַתַּהֲלִיךְ הֵם רַק הִמְשִׁיכוּ אֶת הַשִּׁמּוּשׁ בַּשָּׂפָה "מֵהַמָּקוֹם שֶׁבּוֹ נִפְסְקָה חַיּוּתָהּ", הֲרֵי לַאֲמִתּוֹ שֶׁל דָּבָר הֵם יָצְרוּ מַצָּב לְשׁוֹנִי חָדָשׁ, שֶׁמְּאַפְיְנָיו שְׁאוּבִים גַּם מִן הַשָּׂפָה הָעִבְרִית עַל כָּל תְּקוּפוֹתֶיהָ וְגַם מִן הַשָּׂפוֹת הָאֵירוֹפִּיּוֹת שֶׁעַל רִקְעָן הִתְחוֹלֵל הַתַּהֲלִיךְ, לְמָשָׁל הַיִּידִישׁ (בְּמִדָּה מְסֻיֶּמֶת גַּם מִן הָאֲרָמִית הַתַּלְמוּדִית, הָעֲרָבִית וְשָׂפוֹת נוֹסָפוֹת). הַלָּשׁוֹן הַחֲדָשָׁה שֶׁנּוֹצְרָה הִיא הָעִבְרִית הַיִּשְׂרְאֵלִית בַּת-זְמַנֵּנוּ.

מֵאוֹת מִלִּים חֻדְּשׁוּ בְּאוֹתָהּ תְּקוּפָה, בְּעִקָּר עַל יְדֵי חַיִּים נַחְמָן בְּיָאלִיק, בָּהֶן: תְּגוּבָה, מָטוֹס וּצְרָצַר, וְעַל יְדֵי אֱלִיעֶזֶר בֶּן-יְהוּדָה, בָּהֶן: אֶקְדָּח, חַשְׁמַל, תַּשְׁבֵּץ, גְּלִידָה וְעוֹד
מֵאָז שֶׁנִּפְסַק הַדִּבּוּר בִּלְשׁוֹן חָזָ"ל בַּמֵּאָה הַשְּׁנִיָּה לַסְּפִירָה לֹא הָיְתָה הָעִבְרִית לָשׁוֹן מְדֻבֶּרֶת כִּשְׂפַת אֵם.[2] אַף עַל פִּי כֵן, נַעֲשָׂה בָּהּ בִּתְקוּפַת הַבֵּינַיִם שִׁמּוּשׁ אֵצֶל הַיְּהוּדִים בְּמִגְוָן רָחָב שֶׁל תְּחוּמִים, וְשִׁמּוּשׁ זֶה שֶׁיָּמֵר חֵלֶק נִכְבָּד מֵהַתְּכוּנוֹת שֶׁאִפְיְנוּ אוֹתָהּ.

רֵאשִׁית לַכֹּל – הָעִבְרִית הַקְּלָאסִית הִשְׁתַּמְּרָה בָּאוֹצָר גָּדוֹל יַחֲסִית שֶׁל מְקוֹרוֹת שֶׁהָיוּ מֻכָּרִים הֵיטֵב, בְּרֹאשָׁם הַתָּנָ"ךְ (בְּעִקָּר הַחֲלָקִים שֶׁשִּׁמְּשׁוּ לִצְרָכִים לִיטוּרְגִּיִּים, כְּלוֹמַר הַתּוֹרָה, הַהַפְטָרוֹת, הַמְּגִלּוֹת וְסֵפֶר תְּהִלִּים) וְהַמִּשְׁנָה. מִלְּבַד הַתָּנָ"ךְ וְהַמִּשְׁנָה הָיְתָה הָעִבְרִית שֶׁל הַמְּקוֹרוֹת מֻכֶּרֶת גַּם מֵהַתְּפִלָּה, מֵהַפִּיּוּטִים, מֵהַמִּדְרָשִׁים וּמִשְּׁאַר סִפְרֵי אֲרוֹן הַסְּפָרִים הַיְּהוּדִי שֶׁנִּכְתְּבוּ בְּעִבְרִית.

בְּמַהֲלַךְ יְמֵי הַבֵּינַיִם נַעֲשָׂה שִׁמּוּשׁ בְּעִבְרִית כִּלְשׁוֹן כְּתָב בִּסְפָרוֹת הָרַבָּנִית – שֶׁכָּלְלָה סִפְרֵי פְּסִיקַת הֲלָכָה, שְׁאֵלוֹת וּתְשׁוּבוֹת וְסִפְרֵי הָגוּת, אַךְ גַּם לְצָרְכֵי תַּרְבּוּת כְּגוֹן שִׁירָה וּמְעַט סִפְרוּת. בְּרֹב הַמִּקְרִים, וּבְוַדַּאי בַּתְּקוּפָה וּבָאֵזוֹר שֶׁעַל רִקְעָם צָמַח תַּהֲלִיךְ הַתְּחִיָּה, כְּלוֹמַר בְּאֵירוֹפָּה בַּמֵּאוֹת הַ-18 וְהַ-19, הַשִּׁמּוּשׁ בְּעִבְרִית לֹא הָיָה טִבְעִי, אֶלָּא עָמוּס בִּמְלִיצוֹת וּבְמוּבָאוֹת, בְּצוּרוֹת לֹא-דִּקְדּוּקִיּוֹת וּבַבְּלִילָה שֶׁל לְשׁוֹנוֹת לַעַז, בְּעִקָּר אֲרָמִית. יֶתֶר עַל כֵּן, חֵלֶק מֵהַסִּפְרוּת הָרַבָּנִית לֹא נִכְתַּב כְּלָל בְּעִבְרִית, אֶלָּא בְּאַחַת מִלְּשׁוֹנוֹת הַיְּהוּדִים הָאֲחֵרוֹת, כְּמוֹ אֲרָמִית, עֲרָבִית יְהוּדִית, יִידִישׁ וְכַיּוֹצֵא בְּאֵלֶּה.

מֶחְקָרִים הֶרְאוּ שֶׁעִם הִתְפַּתְּחוּת הַחֲסִידוּת חָלָה הִתְעוֹרְרוּת בַּדִּבּוּר בַּשָּׂפָה הָעִבְרִית, בְּעִקָּר בְּשַׁבָּת, כֵּן יָצְאָה קְרִיאָה לְהַנְחָלַת הַלָּשׁוֹן בְּאֹפֶן נִרְחָב יוֹתֵר. הַוִּכּוּחַ עִם תְּנוּעַת הַהַשְׂכָּלָה הֵבִיא לִירִידַת מְגַמָּה זוֹ וּמִתּוּנָהּ, בַּד בְּבַד עִם הִתְחַזְּקוּת הַמֶּחְקָר, הַכְּתִיבָה וְהַדִּבּוּר הָעִבְרִיִּים בְּקֶרֶב הַמַּשְׂכִּילִים
הַשִּׁמּוּשׁ בְּעִבְרִית לֹא הָיָה רַק בִּלְשׁוֹן כְּתָב. הָעִבְרִית הָיְתָה בְּשִׁמּוּשׁ גַּם כִּלְשׁוֹן הֲגוּיָה, בְּבָתֵּי הַכְּנֶסֶת וּבְבָתֵּי הַמִּדְרָשׁ. בְּאֹפֶן זֶה נִשְׁתַּמְּרוּ כְּלָלֵי הַהֲגִיָּה שֶׁל הָעִבְרִית וְהָאֹפֶן שֶׁבּוֹ מְבַטְּאִים אֶת הָעִצּוּרִים וְהַתְּנוּעוֹת. עִם זֹאת, בְּשֶׁטַח זֶה חָלוּ שִׁנּוּיִים רַבִּים בַּשָּׂפָה בְּהַשְׁפָּעַת לְשׁוֹנוֹת מְקוֹמִיּוֹת, וְהִתְפַּתְּחוּ דַּרְכֵי הֲגִיָּה שׁוֹנוֹת, שֶׁנִּתָּן לְחַלֵּק אוֹתָן בְּאֹפֶן כְּלָלִי לְשָׁלוֹשׁ קְבוּצוֹת:

הַהֲגִיָּה הַסְּפָרַדִּית, שֶׁהָיְתָה בְּשִׁמּוּשׁ אֵצֶל יַהֲדוּת סְפָרַד וְיַהֲדוּת אִיטַלְיָה כְּמוֹ גַּם אֵצֶל רַבִּים מִיְּהוּדֵי אַרְצוֹת הָאִסְלָאם, שֶׁשִּׁמְּרָה מִבְנֶה תְּנוּעָתִי הַשּׁוֹנֶה מִן הַנִּקּוּד הַטַּבְרָנִי הַמֻּכָּר (הָיוּ בָּהּ רַק חָמֵשׁ תְּנוּעוֹת, כְּתוֹצָאָה מֵהִתְמַזְּגוּת הַקָּמָץ עִם הַפֶּתַח וְהַסָּגֹל עִם הַצִּירָה, עַל פִּי הַנִּקּוּד הָאֶרֶצְיִשְׂרְאֵלִי) אַךְ שִׁמְּרָה אֶת הָעִצּוּרִים וְגַם אֶת הַהַטְעָמָה הַדִּקְדּוּקִית וְאֶת הַדְּגֵשִׁים וְהַשְּׁוָאִים;
הַהֲגִיָּה הָאַשְׁכְּנַזִּית, שֶׁשָּׁמְרָה בְּעִקָּר עַל הַהַבְחָנָה בֵּין הַתְּנוּעוֹת (לְפָחוֹת בַּהֲבָרוֹת מֻטְעָמוֹת) אַךְ שִׁנְּתָה אֶת אֹפֶן הַהֲגִיָּה שֶׁל כַּמָּה מֵהֶן, וְכֵן אִבְּדָה חֵלֶק נִכְבָּד מִן הָעִצּוּרִים, אֶת הָאוֹתִיּוֹת הַגְּרוֹנִיּוֹת, וְאֶת הַהַטְעָמָה הַדִּקְדּוּקִית, הַשָּׁוְא הַנָּע וְהַדָּגֵשׁ הֶחָזָק;
הַהֲגִיָּה הַתֵּימָנִית, שֶׁאָמְנָם שִׁמְּרָה כִּמְעַט אֶת כָּל אֹפֶן הַהֲגִיָּה שֶׁל הָעִבְרִית הַקְּלָאסִית אֲבָל לֹא הָיְתָה מֻכֶּרֶת בִּמְקוֹמוֹת שֶׁבָּהֶם הִתְחוֹלֵל תַּהֲלִיךְ הַתְּחִיָּה.
גַּם בְּתוֹךְ כָּל סוּג הָיָה מִגְוָן רָחָב שֶׁל דַּרְכֵי הֲגִיָּה; לְדֻגְמָה, בְּמִסְגֶּרֶת הַהֲגִיָּה הָאַשְׁכְּנַזִּית, הֲגִיַּת יְהוּדֵי פּוֹלִין לֹא הָיְתָה כַּהֲגִיַּת יְהוּדֵי לִיטָא, וּשְׁתֵּיהֶן שׁוֹנוֹת מֵהֲגִיַּת יְהוּדֵי הוּנְגַּרְיָה אוֹ יְהוּדֵי גָּלִיצְיָה, וְגַם בַּהֲגִיַּת הַיִּידִישׁ הָיוּ הֶבְדֵּלִים (לְמָשָׁל: פַּשְׁטִידַת קוּגְל – בְּפוֹלִין, קִיגְל – בְּגָלִיצְיָה, אוֹ: א יוּד (יְהוּדִי) בְּפוֹלִין, א יִיד בְּגָלִיצְיָה).
עַל פִּי עֵדוּיוֹת שֶׁמָּצְאוּ הַחוֹקְרִים מִתְבָּרֵר שֶׁבַּחֲמִשִּׁים הַשָּׁנִים שֶׁקָּדְמוּ לִתְחִלַּת תַּהֲלִיךְ הַתְּחִיָּה כְּבָר הָיָה קַיָּם דִּבּוּר עִבְרִי בְּשׁוּקֵי הָעִיר יְרוּשָׁלַיִם. הַיְּהוּדִים הַסְּפָרַדִּים שֶׁדִּבְּרוּ לְאָדִינוֹ אוֹ עַרְבִית וְהַיְּהוּדִים הָאַשְׁכְּנַזִּים שֶׁדִּבְּרוּ יִידִישׁ נִזְקְקוּ לְשָׂפָה מְשֻׁתֶּפֶת לְצֹרֶךְ מִסְחָר, וּלְצֹרֶךְ זֶה דִּבְּרוּ בֵּינֵיהֶם בְּעִבְרִית. אָמְנָם הָיָה זֶה שִׁמּוּשׁ בְּעִבְרִית כְּשָׂפָה מְדֻבֶּרֶת, וְאוּלָם בָּרוּר שֶׁהָעִבְרִית לֹא שִׁמְּשָׁה כִּשְׂפַת אֵם יְלִידִית אֶלָּא רַק כְּשָׂפָה מְשֻׁתֶּפֶת בִּתְחוּם מְצֻמְצָם, מֵעֵין לִינְגּוֹאָה פִּרְנְקָה שֶׁל הַיְּהוּדִים.

הַמַּצָּב הַלְּשׁוֹנִי שֶׁעַל רִקְעוֹ הִתְרַחֵשׁ תַּהֲלִיךְ הַתְּחִיָּה הָיָה עֲדַיִן מַצָּב שֶׁל דִּיגְלוֹסְיָה, כְּלוֹמַר שֶׁל שִׁמּוּשׁ בִּשְׁתֵּי לְשׁוֹנוֹת, אַחַת לִכְתִיבָה – לְשׁוֹן תַּרְבּוּת, שֶׁנֶּחְשְׁבָה "גְּבוֹהָה" וְאֵיכוּתִית, כְּדֻגְמַת הַלָּטִינִית אוֹ הַיְּוָנִית – וְאַחַת לְדִבּוּר, לָשׁוֹן עֲמָמִית וּ"פְּשׁוּטָה". בְּכָל אַרְצוֹת אֵירוֹפָּה הִתְנַהֵל בַּמֵּאוֹת הָאַחֲרוֹנוֹת תַּהֲלִיךְ שֶׁל יְצִיאָה מִמַּצָּב זֶה, שֶׁרֵאשִׁיתוֹ בְּאַנְגְּלִיָּה בַּמֵּאָה הַ-16, וְאוּלָם עֲדַיִן הַשָּׂפָה שֶׁהָיְתָה מְדֻבֶּרֶת בְּחַיֵּי הַיּוֹם-יוֹם לֹא הָיְתָה לְשׁוֹן הַתַּרְבּוּת וְהַסֵּפֶר; הָרוּסִי דִּבֵּר רוּסִית עֲמָמִית וְכָתַב רוּסִית גְּבוֹהָה אוֹ צָרְפָתִית, הַגֶּרְמָנִי דִּבֵּר בְּנִיב מְקוֹמִי וְכָתַב בְּגֶרְמָנִית תִּקְנִית, וְכֵן הָלְאָה. כָּךְ גַּם אֵצֶל הַיְּהוּדִים, הַלָּשׁוֹן הַמְּדֻבֶּרֶת הָיְתָה יִידִישׁ, וּלְשׁוֹן הַכְּתָב הָיְתָה עִבְרִית לִצְרָכִים דָּתִיִּים, וּלְשׁוֹן הַתַּרְבּוּת הַמְּקוֹמִית – רוּסִית, צָרְפָתִית, גֶּרְמָנִית וְכַדּוֹמֶה – לִצְרָכִים חִלּוֹנִיִּים.
הַתַּהֲלִיךְ שֶׁל תְּחִיַּת הָעִבְרִית נָע לְמַעֲשֶׂה בִּשְׁנֵי קַוִּים מַקְבִּילִים: תְּחִיַּת הָעִבְרִית הַכְּתוּבָה-הַסִּפְרוּתִית וּתְחִיַּת הָעִבְרִית הַמְּדֻבֶּרֶת. בְּעֶשְׂרוֹת הַשָּׁנִים הָרִאשׁוֹנוֹת הַתַּהֲלִיכִים הָאֵלֶּה הָיוּ מְנֻתָּקִים זֶה מִזֶּה וְאַף הִתְרַחֲשׁוּ בִּמְקוֹמוֹת שׁוֹנִים: הָעִבְרִית הַסִּפְרוּתִית הִתְחַדְּשָׁה בְּעָרֵי אֵירוֹפָּה, וְאִלּוּ הָעִבְרִית הַמְּדֻבֶּרֶת הִתְפַּתְּחָה בְּעִקָּר בְּאֶרֶץ יִשְׂרָאֵל (הָיוּ נִסְיוֹנוֹת לְדִבּוּר עִבְרִי גַּם בְּאֵירוֹפָּה, אוּלָם הֵם נִבְלְעוּ בְּמִגְוַן הֶעָצוּם שֶׁל תְּנוּעוֹת אִידֵאוֹלוֹגְיוֹת שֶׁפָּעֲלוּ בְּיַהֲדוּת אֵירוֹפָּה בִּתְחִלַּת הַמֵּאָה הָ-20). הַמִּפְגָּשׁ בֵּין הַתַּהֲלִיכִים הִתְחִיל רַק בְּתוֹךְ הַמֵּאָה הָעֶשְׂרִים, וּנְקֻדַּת צִיּוּן חֲשׁוּבָה בְּעִנְיָן זֶה הִיא עֲלִיָּתוֹ שֶׁל חַיִּים נַחְמָן בְּיָאלִיק לְאֶרֶץ יִשְׂרָאֵל בְּ-1924. גַּם לְאַחַר הַמַּעֲבָר שֶׁל הָעִבְרִית הַסִּפְרוּתִית לָאָרֶץ נִשְׁאַר הֶבְדֵּל נִכָּר בֵּין מְאַפְיְנֵי הָעִבְרִית הַסִּפְרוּתִית וְהַמְּדֻבֶּרֶת, וְהַהֶבְדֵּל הַזֶּה קַיָּם גַּם הַיּוֹם; מְאַפְיְנִים שֶׁל דִּבּוּר עִבְרִי חָדְרוּ לִלְשׁוֹן הַדִּיאָלוֹג בַּסִּפְרוּת רַק הָחֵל מִשְּׁנוֹת הָ-40 שֶׁל הַמֵּאָה הָ-20 (בְּסִפְרֵיהֶם שֶׁל מֹשֶׁה שָׁמִיר וְס' יִזְהָר), וְלִלְשׁוֹן הַנָּרָטִיב בַּסִּפְרוּת רַק בִּשְׁנוֹת הַ-90 שֶׁל הַמֵּאָה הָ-20 (מְפֻרְסָמִים בְּהֶקְשֵׁר זֶה סְפָרָיו שֶׁל אֶתְגָּר קֶרֶת).
תַּהֲלִיךְ מַקְדִּים לִתְחִיַּת הָעִבְרִית הַסִּפְרוּתִית הִתְרַחֵשׁ בַּסִּפְרוּת שֶׁל תְּנוּעַת הַהַשְׂכָּלָה בְּאֵירוֹפָּה. הַמַּשְׂכִּילִים שֶׁבִּקְּשׁוּ לְהִתְרַחֵק מִן הַתַּרְבּוּת הָרַבָּנִית וְהַיִּידִישָׁאִית שֶׁכְּנֶגְדָּהּ פָּעֲלוּ, הֶחֱלִיטוּ שֶׁסִּפְרוּת יָפֶה רְאוּיָה לְהִכָּתֵב בְּעִבְרִית, וְדַוְקָא בְּעִבְרִית מִקְרָאִית. סוּגֵי הַלָּשׁוֹן הָאֲחֵרִים שֶׁשִּׁמְּשׁוּ אֶת הַיְּהוּדִים נִרְאוּ בְּעֵינֵיהֶם מְשֻׁבָּשִׁים וְלֹא רְאוּיִים לִכְתִיבָה, וּבֵינֵיהֶם לְשׁוֹן חָזָ"ל הַלָּשׁוֹן הָאֲרָמִית-יְהוּדִית וְיִידִישׁ שֶׁבָּהּ עֵרוּב שֶׁל עִבְרִית, גֶּרְמָנִית-פְרֶנְקִית, וְשָׂפוֹת סְלָבִיּוֹת. הַסִּפְרוּת הַמַּשְׂכִּילִית שֶׁנִּכְתְּבָה בְּעִבְרִית הִתְבַּסְּסָה עַל שְׁנֵי עֶקְרוֹנוֹת מֶרְכָּזִיִּים: טַהֲרָנוּת לְשׁוֹנִית (פּוּרִיזְם) וּמְלִיצָה. פּוּרִיזְם הָיָה עִקָּרוֹן שֶׁלְּפִיו בַּטֵּקְסְט שֶׁנִּכְתַּב הוֹפִיעוּ אַךְ וְרַק מִלִּים שֶׁמְּקוֹרָן בַּמִּקְרָא (גַּם אִם בְּמַשְׁמָעוּת לֹא מִקְרָאִית). עֶקְרוֹן הַמְּלִיצָה הִתְבַּסֵּס עַל הֲבָאַת פְּסוּקִים וּבִטּוּיִים שְׁלֵמִים כְּצוּרָתָם מִן הַמִּקְרָא, וּכְכָל שֶׁטֵּקְסְט סִפְרוּתִי הֵכִיל יוֹתֵר מְלִיצוֹת כָּךְ נֶחֱשָׁב אֵיכוּתִי יוֹתֵר. מְאַפְיֵן לְשׁוֹנִי נוֹסָף שֶׁהֶעֱלָה אֶת קַרְנוֹ שֶׁל הַטֵּקְסְט הָיָה שִׁמּוּשׁ מְרֻבֶּה בְּמִלִּים יְחִידָאִיּוֹת. כְּדֻגְמָה לְטֵקְסְט מַשְׂכִּילִי, הִנֵּה פִּסְקַת הַפְּתִיחָה שֶׁל הָרוֹמָן "אַהֲבַת צִיּוֹן"[7] מֵאֵת אַבְרָהָם מַאפּוּ:
לְצַד הַסִּפְרוּת הֵקִימוּ אַנְשֵׁי תְּנוּעַת הַהַשְׂכָּלָה עוֹלָם שֶׁל עִתּוֹנוּת עִבְרִית. הָחֵל מִ"קֹהֶלֶת מוּסָר", נִסְיוֹן הַבֹּסֶר שֶׁל מֹשֶׁה מֶנְדֶּלְסוֹן בִּשְׁנוֹת הַ-50 שֶׁל הַמֵּאָה הַ-18, דֶּרֶךְ "הַמְּאַסֵּף", "בִּכּוּרֵי הָעִתִּים" וְ"כֶּרֶם חֶמֶד", וְעַד "הַמַּגִּיד", "הַמֵּלִיץ", "הֶחָלוּץ", "הַצְּפִירָה", "הַשַּׁחַר" וַאֲחֵרִים. הַמַּשְׂכִּילִים שָׁמְרוּ עַל הַקְּשָׁרִים בֵּינֵיהֶם בְּאֶמְצָעוּת כִּתְבֵי עֵת וְעִתּוֹנִים עִבְרִיִּים שֶׁהוּפְצוּ בְּרַחֲבֵי הַקְּהִלּוֹת בְּאֵירוֹפָּה וְכָלְלוּ פּוּבְּלִיצִיסְטִיקָה, מַאֲמָרִים בְּחָכְמַת יִשְׂרָאֵל וְקִטְעֵי סִפְרוּת וְשִׁירָה. כִּתְבֵי עֵת אֵלֶּה תָּרְמוּ תְּרוּמָה חֲשׁוּבָה לְהִתְחַדְּשׁוּת הָעִבְרִית וּלְהַרְחָבַת הַשִּׁמּוּשׁ בַּשָּׂפָה לְזָ'אנֵרִים נוֹסָפִים.

הַכְּבָלִים הַלְּשׁוֹנִיִּים שֶׁבָּהֶם כָּבְלָה אֶת עַצְמָהּ הַסִּפְרוּת הַמַּשְׂכִּילִית הִקְשׁוּ יוֹתֵר וְיוֹתֵר עַל כְּתִיבָה חָפְשִׁית וְרַחֲבַת הֶקֵּף. אִם בְּ"אַהֲבַת צִיּוֹן", רוֹמָן הַמִּתְרַחֵשׁ בְּרֶקַע מִקְרָאִי, הִצְלִיחַ מַאפּוּ לָתֵת תֵּאוּרִים סְבִירִים שֶׁל הַדְּמוּיוֹת, הַמְּקוֹמוֹת וְהָעֲלִילָה, הֲרֵי כְּשֶׁתֵּאֵר בְּ"עַיִט צָבוּעַ" אֶת הַטִּפּוּסִים בְּיַהֲדוּת לִיטָא שֶׁל זְמַנּוֹ נֵאָלֵץ לַעֲשׂוֹת שִׁמּוּשׁ גַּם בִּלְשׁוֹן חָזָ"ל. כְּשֶׁפָּנוּ סוֹפְרֵי הַהַשְׂכָּלָה לַעֲסֹק בַּעֲלִילוֹת הַמִּתְרַחֲשׁוֹת בִּזְמַנִּים מוֹדֶרְנִיִּים יוֹתֵר הֻרְגַּשׁ יוֹתֵר וְיוֹתֵר הַמַּחְסוֹר בְּאוֹצַר מִלִּים רָחָב מַסְפִּיק לְצֹרֶךְ הַתֵּאוּר; קַל וָחֹמֶר כְּשֶׁבִּקְּשׁוּ לִכְתֹּב סִפְרֵי מַדָּע וְטֶבַע אוֹ לְתַרְגֵּם סִפְרוּת יָפָה מִלְּשׁוֹנוֹת אֵירוֹפָּה. אֶת הַמַּחְסוֹם שֶׁנּוֹצַר פָּרַץ, בִּשְׁנוֹת הַ-80 שֶׁל הַמֵּאָה הַ-19, מֶנְדֵּלֵי מוֹכֵר סְפָרִים.
מֶנְדֵּלֵי (שָׁלוֹם יַעֲקֹב אַבְרָמוֹבִיץ, 1836–1917) הֵחֵל אֶת כְּתִיבָתוֹ בְּעִבְרִית כְּסוֹפֵר מַשְׂכִּילִי וְכָתַב סִפְרוּת לְפִי כָּל כְּלָלֵי הָעִבְרִית הַמַּשְׂכִּילִית. בְּשָׁלָב מְסֻיָּם פָּנָה לִכְתֹּב בְּיִידִישׁ וְעָרַךְ גַּם בְּשָׂפָה זוֹ מַהְפֵּכָה לְשׁוֹנִית, שֶׁהִתְבַּטְּאָה בְּהַרְחָבַת הַשִּׁמּוּשׁ בְּיִידִישׁ הַמְּדֻבֶּרֶת לְצָרְכֵי סִפְרוּת יָפָה. אַחֲרֵי הַפְסָקָה גְּדוֹלָה חָזַר בְּ-1886 לִכְתֹּב בְּעִבְרִית, אֲבָל הֶחְלִיט לְוַתֵּר עַל הַשִּׁמּוּשׁ בִּכְלָלֵי הָעִבְרִית הַמִּקְרָאִית וְהִכְנִיס לְתוֹךְ אוֹצַר הַמִּלִּים וְהַדִּקְדּוּק שֶׁלּוֹ אֶת לְשׁוֹן חָזָ"ל וְהַלָּשׁוֹן הָרַבָּנִית שֶׁל יְמֵי הַבֵּינַיִם. בְּסִפְרוֹ "בְּסֵתֶר רַעַם" שֶׁיָּצָא לָאוֹר בְּ-1886 פָּתַח סִגְנוֹן חָדָשׁ בַּכְּתִיבָה הָעִבְרִית, סִגְנוֹן קוֹלֵחַ וּמְגֻוָּן הַמְּשַׁקֵּף אֶת הַיִּידִישׁ שֶׁדֻּבְּרָה בִּסְבִיבָתוֹ, וּמִשְׁתַּמֵּשׁ בְּכָל הָרְבָדִים הַהִיסְטוֹרִיִּים שֶׁל הָעִבְרִית. לְצֹרֶךְ סְפָרָיו הָעִבְרִיִּים, שֶׁהָיוּ בְּחֶלְקָם תַּרְגּוּם שֶׁל הַסְּפָרִים שֶׁכָּתַב בְּיִידִישׁ, נִזְקַק לְלָשׁוֹן שֶׁתְּיַצֵּג דִּבּוּר עֲמָמִי שׁוֹטֵף, מְתֻבָּל בַּהֲלָצוֹת וּבְתֵאוּרִים מְפֹרָטִים, וְאֶת הַצֹּרֶךְ הַזֶּה סִפֵּק בְּהִשְׁתַּחְרְרוּת מִכַּבְלֵי הַמְּלִיצָה הַמִּקְרָאִית הַמַּשְׂכִּילִית, בִּפְנִיָּה לְמַטְבְּעוֹת לָשׁוֹן וְאוֹצַר מִלִּים חז"ליים וּבְשִׁלּוּב מְאַפְיְנִים תַּחְבִּירִיִּים מִן הַשָּׂפוֹת הָאֵירוֹפִּיּוֹת.

שְׂפָתוֹ שֶׁל מֶנְדֵּלֵי נִתְפְּסָה כְּ"לָשׁוֹן סִינְתֵּטִית", כְּלוֹמַר מֻרְכֶּבֶת מִכַּמָּה רָבְדֵי לָשׁוֹן קוֹדְמִים, וְלֹא כְּהֶמְשֵׁךְ יָשִׁיר שֶׁל רֹבֶד לָשׁוֹן מְסֻיָּם. בַּזְּמַן הָאַחֲרוֹן עָבְרוּ הַחוֹקְרִים לִתְפִיסָה שֶׁלְּשׁוֹנוֹ מַמְשִׁיכָה בְּהֶמְשֵׁךְ יָשִׁיר אֶת הַסִּפְרוּת הָרַבָּנִית וְהַחֲסִידִית שֶׁל תְּחִלַּת הָעֵת הַחֲדָשָׁה, עִם הוֹסָפָה שֶׁל יְסוֹדוֹת חֲדָשִׁים, בְּעִקָּר דִּקְדּוּקִיִּים.

הֶמְשֵׁךְ תַּהֲלִיךְ הַתְּחִיָּה הַסִּפְרוּתִית
הַסִּגְנוֹן שֶׁל מֶנְדֵּלֵי אֻמַּץ בְּהִתְלַהֲבוּת עַל יְדֵי סוֹפְרֵי הַתְּקוּפָה וְהִתְפַּשֵּׁט בִּמְהִירוּת. הוּא הֻרְחַב גַּם לִתְחוּמִים נוֹסָפִים: אַחַד הָעָם כָּתַב בְּ-1889 מַאֲמָר בְּשֵׁם "לֹא זֶה הַדֶּרֶךְ" שֶׁנֶּחְשַׁב כְּמוֹפֵת לִכְתִיבָה עִיּוּנִית בַּסִּגְנוֹן הֶחָדָשׁ, וּבְיָאלִיק הִרְחִיב, עִם פִּרְסוּם שִׁירוֹ הָרִאשׁוֹן "אֶל הַצִּפּוֹר" בְּ-1889, אֶת הַשִּׁמּוּשׁ בַּסִּגְנוֹן הֶחָדָשׁ לְצֹרֶךְ כְּתִיבַת שִׁירָה. נַעֲשָׂה מַאֲמָץ גָּדוֹל לִכְתֹּב גַּם סִפְרֵי טֶבַע וּמַדָּע בְּעִבְרִית, וּלְשֵׁם כָּךְ אַף הֻרְחַב אוֹצַר הַמִּלִּים הַטֶּכְנִי וְהַמַּדָּעִי. בְּאֵירוֹפָּה נוֹסְדוּ עִתּוֹנִים וְכִתְבֵי עֵת בַּשָּׂפָה הָעִבְרִית, וִישִׁיבוֹת וְדִיּוּנִים שֶׁל כְּנָסִים צִיּוֹנִיִּים וַאֲחֵרִים נֶעֶרְכוּ וְנִרְשְׁמוּ בְּעִבְרִית. הַכְּתִיבָה הָעִבְרִית הָלְכָה וְהִתְרַחֲבָה, וְעִם מַעֲבַר הַסּוֹפְרִים וְהַמְּשׁוֹרְרִים לְאֶרֶץ יִשְׂרָאֵל, בְּעִקָּר בָּעֲלִיָּה הַשְּׁנִיָּה, הִשְׁפִּיעָה הָעִבְרִית הַסִּפְרוּתִית הַחֲדָשָׁה בְּמִדָּה מְסֻיֶּמֶת גַּם עַל תַּהֲלִיךְ תְּחִיַּת הַדִּבּוּר שֶׁהִתְנַהֵל בְּאוֹתָהּ תְּקוּפָה בָּאָרֶץ.

עַל אַף הַהִתְקָרְבוּת הַגֵּאוֹגְרָפִית וְהַלְּשׁוֹנִית בֵּין הַתַּהֲלִיכִים, נִשְׁאֲרָה הָעִבְרִית הַכְּתוּבָה שׁוֹנָה בִּמְאַפְיְנִים רַבִּים מִן הָעִבְרִית הַמְּדֻבֶּרֶת, יוֹתֵר מֵאֲשֶׁר בְּשָׂפוֹת אֲחֵרוֹת שֶׁיֵּשׁ לָהֶן לְשׁוֹן דִּבּוּר וּלְשׁוֹן כְּתָב. אוֹצַר הַמִּלִּים, הַדִּקְדּוּק וְהַתַּחְבִּיר הָעִבְרִיִּים מִתְנַהֲגִים אַחֶרֶת בְּמִסְמָךְ כָּתוּב וּבְדִבּוּר זוֹרֵם. גַּם הַיִּצּוּג שֶׁל הָעִבְרִית הַמְּדֻבֶּרֶת בַּסִּפְרוּת הָיָה בַּתְּחִלָּה בְּסִגְנוֹן גָּבוֹהַּ וְרִשְׁמִי מְאוֹד, וְרַק בַּמַּחֲצִית הַשְּׁנִיָּה שֶׁל הַמֵּאָה הָעֶשְׂרִים חָדְרוּ מְאַפְיְנִים סִגְנוֹנִיִּים שֶׁל הָעִבְרִית הַמְּדֻבֶּרֶת לְתוֹךְ לְשׁוֹן הַדִּיאָלוֹג הַסִּפְרוּתִי.
